NeonFast and Safe Native Node.js Modules

// JS
function hello() {
  let result = fibonacci(10000);
  console.log(result);
  return result;
}


// Neon
fn hello(mut cx: FunctionContext) -> JsResult<JsNumber> {
  let result = cx.number(fibonacci(10000));
  println!("{}", result);
  Ok(result)
}
